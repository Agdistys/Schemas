  <script>

    // ========= CONFIG =========

    const OWNER = "agdistys";               // ton pseudo GitHub

    const REPO  = "agdistys.github.io";     // ton dépôt Pages (user site)

    // Si tes images sont dans un sous-dossier (ex. "Schemas"), mets BASE_PATH = "Schemas".

    // Sinon laisse "" pour la racine.

    const BASE_PATH = ""; // ex: "Schemas" ou "" si tes dossiers sont à la racine

 

    // Dossiers à afficher (adapter aux noms EXACTS que tu vois dans le dépôt)

    const FOLDERS = [

      "", // fichiers à la racine (hors dossier)

      "ADDICTO",

      "CANCER",

      "COGNITIF",

      "Cartes psycho-emotionnelles",

      "EGO & ECO",

      "LOIS",

      "Poesies",

      "SOLO",

      "Societe",

      "Tableaux PsychoBiologie",

      "Ups & Downs",

      "Yin Yang"

    ];

 

    const IMAGE_EXT = ["png","jpg","jpeg","webp","gif","svg"]; // extensions affichées

    // ==========================

 

    const apiBase = `https://api.github.com/repos/${OWNER}/${REPO}/contents`;

    // URL publique pour afficher les images via GitHub Pages :

    // si BASE_PATH = "Schemas" et file = "ADDICTO/img.png" => /Schemas/ADDICTO/img.png

    const publicUrl = (relativePath) => {

      const prefix = BASE_PATH ? `/${encodeURI(BASE_PATH)}` : "";

      return `${prefix}/${relativePath.split("/").map(encodeURI).join("/")}`;

    };

 

    async function listPath(path) {

      // path peut être "", "ADDICTO", "ADDICTO/sousdossier", etc.

      const full = [BASE_PATH, path].filter(Boolean).map(encodeURIComponent).join("/");

      const url = `${apiBase}/${full}`;

      const res = await fetch(url, { headers: { "Accept":"application/vnd.github.v3+json" }});

      if (!res.ok) throw new Error(`${path||"Racine"} → ${res.status} ${res.statusText}`);

      return res.json(); // array of {name, path, type, download_url, html_url, ...}

    }

 

    function isImage(name) {

      const ext = name.split(".").pop().toLowerCase();

      return IMAGE_EXT.includes(ext);

    }

 

    function section(title) {

      const sec = document.createElement("section");

      sec.className = "sec";

      sec.innerHTML = `<h2>${title || "Racine"}</h2><div class="grid"></div>`;

      return sec;

    }

 

    async function renderFolder(rootEl, folder) {

      const sec = section(folder);

      const grid = sec.querySelector(".grid");

      rootEl.appendChild(sec);

 

      try {

        const entries = await listPath(folder);

        const files = entries.filter(x => x.type === "file" && isImage(x.name));

        if (files.length === 0) {

          const p = document.createElement("p");

          p.className = "empty";

          p.textContent = "Aucune image trouvée ici pour le moment.";

          sec.appendChild(p);

          return;

        }

        for (const f of files) {

          const relPath = (folder ? `${folder}/${f.name}` : f.name);

          const url = publicUrl(relPath); // URL servie par Pages

          const card = document.createElement("div");

          card.className = "card";

          card.innerHTML = `

            <a href="${url}" target="_blank" rel="noopener">

              <img class="thumb" loading="lazy" src="${url}" alt="${f.name}">

              <div class="cap">${f.name}</div>

            </a>

          `;

          grid.appendChild(card);

        }

      } catch (e) {

        const err = document.createElement("div");

        err.className = "err";

        err.textContent = "Erreur de listing : " + e.message + "\n→ Vérifie le chemin/dossier et qu’il existe bien dans le dépôt.";

        sec.appendChild(err);

      }

    }

 

    async function renderAll() {

      const root = document.getElementById("gallery");

      root.innerHTML = "";

      for (const folder of FOLDERS) {

        await renderFolder(root, folder);

      }

    }

 

    renderAll();

  </script>
